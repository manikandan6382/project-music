<script>
document.addEventListener('scroll', function() {
  var scrollPosition = window.scrollY;
  var fixedBox = document.querySelector('.fixed-box');
  
  if (scrollPosition > 0) {
    fixedBox.classList.add('show');
  } else {
    fixedBox.classList.remove('show');
  }
});

</script>
